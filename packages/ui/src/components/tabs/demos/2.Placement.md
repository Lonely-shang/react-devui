# en-US

Modify the header position.

# zh-Hant

修改页头位置。

```tsx
import { useState } from 'react';

import { DTabs, DRadioGroup } from '@react-devui/ui';

export default function Demo() {
  const [placement, setPlacement] = useState('top');

  return (
    <>
      <DRadioGroup
        dOptions={['top', 'right', 'bottom', 'left'].map((placement) => ({
          label: placement,
          value: placement,
        }))}
        dModel={[placement, setPlacement]}
        dType="outline"
      />
      <br />
      <DTabs
        dTabs={[1, 22, 333, 4444].map((n) => ({
          id: n,
          title: `Tab${n}`,
          panel: `Tab${n} Content...`,
          disabled: n === 333,
        }))}
        dPlacement={placement}
      />
    </>
  );
}
```
