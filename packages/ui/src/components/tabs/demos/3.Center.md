# en-US

The label is displayed in the center.

# zh-Hant

标签居中展示。

```tsx
import { useState } from 'react';

import { DTabs, DRadioGroup } from '@react-devui/ui';
import { AppstoreOutlined } from '@react-devui/ui/icons';

export default function Demo() {
  const [placement, setPlacement] = useState('top');

  return (
    <>
      <DRadioGroup
        dOptions={['top', 'right', 'bottom', 'left'].map((placement) => ({
          label: placement,
          value: placement,
        }))}
        dModel={[placement, setPlacement]}
        dType="outline"
      />
      <br />
      <DTabs
        dTabs={[1, 22, 333, 4444].map((n) => ({
          id: n,
          title: (
            <>
              <AppstoreOutlined style={{ marginRight: 4 }} />
              Tab {n}
            </>
          ),
          panel: `Tab${n} Content...`,
          disabled: n === 333,
        }))}
        dPlacement={placement}
        dCenter
      />
    </>
  );
}
```
