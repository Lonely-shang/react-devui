# zh-Hant

最简单的用法。

```tsx
import { useState } from 'react';

import { DSelect } from '@react-devui/ui';

const options = Array(100)
  .fill(0)
  .map((item, index) => ({
    label: `Option ${index}`,
    value: index,
    disabled: index === 3,
  }));
export default function Demo() {
  const [select, setSelect] = useState(50);

  return (
    <>
      <DSelect dOptions={options} placeholder="Basic" />
      <DSelect dOptions={options} placeholder="Disabled" disabled />
      <DSelect dOptions={options} placeholder="Clearable" dModel={[select, setSelect]} dClearable />
    </>
  );
}
```
