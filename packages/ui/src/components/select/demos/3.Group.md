# en-US

Group options.

# zh-Hant

对选项进行分组。

```tsx
import { DSelect } from '@react-devui/ui';

const options = [
  {
    label: 'Group 1',
    value: 'Group1',
    children: Array(3)
      .fill(0)
      .map((item, index) => ({
        label: `Option ${index}`,
        value: 'Group1-' + index,
      })),
  },
  {
    label: 'Group 2',
    value: 'Group2',
    children: [],
  },
  {
    label: 'Group 3',
    value: 'Group3',
    children: Array(100)
      .fill(0)
      .map((item, index) => ({
        label: `Option 3${index}`,
        value: 'Group3-' + index,
        disabled: index === 3,
      })),
  },
];
export default function Demo() {
  return <DSelect dOptions={options} dPlaceholder="Group" dSearchable dClearable />;
}
```
