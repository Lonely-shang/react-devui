# en-US

Adjust the size by setting `dSize` to `larger` and `smaller`.

# zh-Hant

通过设置 `dSize` 为 `larger` `smaller` 调整尺寸。

```tsx
import { useState } from 'react';

import { DSelect, DRadioGroup } from '@react-devui/ui';

const options = Array(100)
  .fill(0)
  .map((item, index) => ({
    label: `Option ${index}`,
    value: index,
    disabled: index === 3,
  }));
export default function Demo() {
  const [_size, setSize] = useState('default');
  const [select, setSelect] = useState([10, 20]);

  const size = _size === 'default' ? undefined : _size;

  return (
    <>
      <DRadioGroup
        dOptions={['smaller', 'default', 'larger'].map((size) => ({
          label: size,
          value: size,
        }))}
        dModel={[_size, setSize]}
        dType="outline"
      />
      <br />
      <DSelect dOptions={options} dPlaceholder="Basic" dSize={size} />
      <DSelect dOptions={options} dPlaceholder="Multiple" dModel={[select, setSelect]} dMultiple dSearchable dClearable dSize={size} />
    </>
  );
}
```
