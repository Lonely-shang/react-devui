# en-US

When `dCreateOption` is set, the creation behavior is determined by the return value.

# zh-Hant

当设置了 `dCreateOption`，由返回值决定创建行为。

```tsx
import { useState } from 'react';

import { DSelect } from '@react-devui/ui';

export default function Demo() {
  const [options, setOptions] = useState(() =>
    Array(100)
      .fill(0)
      .map((item, index) => ({
        label: `Option ${index}`,
        value: index,
        disabled: index === 3,
      }))
  );

  const createFn = (value) => {
    if (/^[0-9]+$/.test(value) && Number(value) > options.length) {
      return {
        label: `Option ${value}`,
        value: Number(value),
      };
    }
  };

  const handleCreateOption = (option) => {
    setOptions([option, ...options]);
  };

  return (
    <>
      <DSelect dOptions={options} placeholder="Create" dSearchable dCreateOption={createFn} onCreateOption={handleCreateOption} />
      <DSelect
        dOptions={options}
        placeholder="Create in multiple mode"
        dSearchable
        dCreateOption={createFn}
        onCreateOption={handleCreateOption}
        dMultiple
      />
    </>
  );
}
```
