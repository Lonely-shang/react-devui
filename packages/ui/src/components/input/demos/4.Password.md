# en-US

Hide the specific content entered.

# zh-Hant

隐藏输入的具体内容。

```tsx
import { useState } from 'react';

import { DInput } from '@react-devui/ui';

export default function Demo() {
  const [value1, setValue1] = useState('Password');
  const [value2, setValue2] = useState('Password');

  return (
    <>
      <DInput dModel={[value1, setValue2]} dPlaceholder="Password" dType="password"></DInput>
      <DInput dModel={[value2, setValue2]} dPlaceholder="Password" dType="password" dClearable></DInput>
      <DInput dModel={[value2, setValue2]} dPlaceholder="Password" dType="password" dClearable dDisabled></DInput>
    </>
  );
}
```
