# en-US

Let the user enter numbers.

# zh-Hant

让用户输入数字。

```tsx
import { useState } from 'react';

import { DInput } from '@react-devui/ui';

export default function Demo() {
  const [value, setValue] = useState('123');

  return (
    <>
      <DInput placeholder="Number" type="number" />
      <DInput placeholder="Number" type="number" dModel={[value, setValue]} dClearable />
      <DInput placeholder="max, min, step" type="number" max={1000} min={10} step={10} dPrefix="$" />
      <DInput placeholder="Decimal values" type="number" max={100} min={0} step={0.01} dSuffix="%" />
      <DInput placeholder="Decimal values" type="number" max={100} min={0} step={0.01} dSuffix="%" disabled />
    </>
  );
}
```
