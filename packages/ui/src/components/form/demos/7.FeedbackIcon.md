# en-US

通过 `dFeedbackIcon` 来配置反馈图标。

# zh-Hant

通过 `dFeedbackIcon` 来配置反馈图标。

```tsx
import { DForm, DFormItem, FormControl, FormGroup, Validators, useForm, DInput } from '@react-devui/ui';
import { CloseCircleOutlined, WarningOutlined } from '@react-devui/ui/icons';

export default function Demo() {
  const formInstance1 = useForm(
    () =>
      new FormGroup({
        username: new FormControl('', [Validators.required, Validators.minLength(5), Validators.maxLength(12)]),
      })
  );
  const formInstance2 = useForm(
    () =>
      new FormGroup({
        username: new FormControl('', [Validators.required, Validators.minLength(5), Validators.maxLength(12)]),
      })
  );

  return (
    <>
      <DForm dForm={formInstance1} dLabelWidth={120} dFeedbackIcon>
        <DFormItem
          dFormControls={{
            username: {
              required: 'Please input your username!',
              minLength: {
                message: 'Minimum length is 5!',
                status: 'warning',
              },
              maxLength: 'Maximum length is 12!',
            },
          }}
          dLabel="Default"
        >
          {({ username }) => <DInput dFormControl={username} placeholder="required, minLength:warning,5, maxLength:12" />}
        </DFormItem>
      </DForm>
      <DForm
        dForm={formInstance2}
        dLabelWidth={120}
        dFeedbackIcon={{
          error: <CloseCircleOutlined />,
          warning: <WarningOutlined />,
        }}
      >
        <DFormItem
          dFormControls={{
            username: {
              required: 'Please input your username!',
              minLength: {
                message: 'Minimum length is 5!',
                status: 'warning',
              },
              maxLength: 'Maximum length is 12!',
            },
          }}
          dLabel="Custom"
        >
          {({ username }) => <DInput dFormControl={username} placeholder="required, minLength:warning,5, maxLength:12" />}
        </DFormItem>
      </DForm>
    </>
  );
}
```
