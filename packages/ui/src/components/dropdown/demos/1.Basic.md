# en-US

The simplest usage.

Automatically adjust the position beyond the `window` view.

# zh-Hant

最简单的用法。

超出 `window` 视图自动调整位置。

```tsx
import { DDropdown, DDropdownItem, DDropdownGroup, DButton, DIcon } from '@react-devui/ui';

export default function Demo() {
  const icon1 = <DIcon dName="arrow-down"></DIcon>;

  const icon2 = <DIcon dName="experiment"></DIcon>;

  return (
    <>
      <DDropdown
        dTriggerNode={
          <DButton dType="text" dIcon={icon1} dIconRight>
            Hover me
          </DButton>
        }
      >
        <DDropdownItem dId="Item1">Item 1</DDropdownItem>
        <DDropdownItem dId="Item2" dDisabled>
          Item 2
        </DDropdownItem>
        <DDropdownItem dId="Item3" dIcon={icon2}>
          Item 3
        </DDropdownItem>
      </DDropdown>
      <DDropdown
        dTriggerNode={
          <DButton dType="secondary" dIcon={icon1} dIconRight>
            Click me
          </DButton>
        }
        dTrigger="click"
      >
        <DDropdownItem dId="Item1">Item 1</DDropdownItem>
        <DDropdownItem dId="Item2">Item 2</DDropdownItem>
        <DDropdownGroup dTitle="Group 3">
          <DDropdownItem dId="Item31">Item 31</DDropdownItem>
          <DDropdownItem dId="Item32">Item 32</DDropdownItem>
        </DDropdownGroup>
        <DDropdownItem dId="Item4">Item 4</DDropdownItem>
      </DDropdown>
    </>
  );
}
```
