# en-US

The checkbox group reveals the checked state.

# zh-Hant

多选组暴露出了选中状态。

```tsx
import { useState, useCallback, useRef } from 'react';

import { DCheckbox, DCheckboxGroup } from '@react-devui/ui';
import { useRefCallback } from '@react-devui/ui/hooks';

export default function Demo() {
  const [data, ref] = useRefCallback();
  const [value, setValue] = useState([2]);

  return (
    <>
      {data && <DCheckbox {...data.indeterminateProps}>{data.indeterminateChecked === true ? 'Clear all' : 'Check all'}</DCheckbox>}
      <DCheckboxGroup ref={ref} dModel={[value, setValue]}>
        {[1, 2, 3].map((n) => (
          <DCheckbox key={n} dValue={n}>
            Checkbox {n}
          </DCheckbox>
        ))}
      </DCheckboxGroup>
    </>
  );
}
```
