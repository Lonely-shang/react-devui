# en-US

The most basic drop.

# zh-Hant

最基本的放置。

```tsx
import { DDrag, DDrop, DDragPlaceholder, DButtonGroup, DButton } from '@react-devui/ui';
import { useImmer, useRefCallback } from '@react-devui/ui/hooks';

export default function Demo() {
  const [dropEl, dropRef] = useRefCallback<HTMLLIElement>();

  const [data, setData] = useImmer([1, 2, 3]);

  return (
    <>
      <DButtonGroup>
        <DButton
          onClick={() =>
            setData((draft) => {
              draft.unshift(draft.length + 1);
            })
          }
        >
          Add top
        </DButton>
        <DButton
          onClick={() =>
            setData((draft) => {
              draft.splice(~~(draft.length / 2), 0, draft.length + 1);
            })
          }
        >
          Add center
        </DButton>
        <DButton
          onClick={() =>
            setData((draft) => {
              draft.push(draft.length + 1);
            })
          }
        >
          Add end
        </DButton>
        <DButton onClick={() => setData(data.filter((n) => n <= 3))}>Clear</DButton>
      </DButtonGroup>
      <br />
      <br />
      <div ref={dropRef} className="app-demo-drop">
        <DDrop
          dList={[data, setData]}
          dItemRender={(item) => <div className="app-demo-drag">Drag {item}</div>}
          dGetId={(item) => String(item)}
          dContainer={dropEl}
          dPlaceholder={<DDragPlaceholder className="app-demo-drag-placeholder" />}
        ></DDrop>
      </div>
    </>
  );
}
```
