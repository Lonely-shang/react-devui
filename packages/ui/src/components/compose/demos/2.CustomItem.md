# en-US

Provide any combinable items through `DComposeItem`.

# zh-Hant

通过 `DComposeItem` 提供任意可组合的项。

```tsx
import { useState } from 'react';

import { DCompose, DComposeItem, DIcon, DInput, DInputAffix, DSelect, DCascader } from '@react-devui/ui';

export default function Demo() {
  const settingIcon = <DIcon dName="setting"></DIcon>;
  const [options] = useState(() =>
    Array(100)
      .fill(0)
      .map((item, index) => ({ dLabel: `react-devui ${index + 1}`, dValue: index + 1, dDisabled: index === 3 }))
  );
  const [cascaderOptions] = useState(() =>
    Array(3)
      .fill(0)
      .map((item, index1) => ({
        dLabel: `Option ${index1 + 1}`,
        dValue: index1 + 1,
        dChildren: Array(5)
          .fill(0)
          .map((item, index2) => ({
            dLabel: `Option ${index1 + 1}${index2 + 1}`,
            dValue: index2 + 1,
            dDisabled: index2 === 3,
            dChildren: Array(20)
              .fill(0)
              .map((item, index3) => ({
                dLabel: `Option ${index1 + 1}${index2 + 1}${index3 + 1}`,
                dValue: index3 + 1,
              })),
          })),
      }))
  );

  return (
    <>
      <DCompose>
        <DInput placeholder="Setting" />
        <DComposeItem>{settingIcon}</DComposeItem>
      </DCompose>
      <br />
      <DCompose>
        <DComposeItem dGray>http://</DComposeItem>
        <DInput placeholder="URL" />
        <DComposeItem dGray>.com</DComposeItem>
      </DCompose>
      <br />
      <DCompose>
        <DComposeItem dGray>http://</DComposeItem>
        <DSelect dOptions={options} dPlaceholder="URL"></DSelect>
        <DComposeItem dGray>.com</DComposeItem>
      </DCompose>
      <br />
      <DCompose>
        <DComposeItem dGray>http://</DComposeItem>
        <DCascader dOptions={cascaderOptions} dPlaceholder="URL"></DCascader>
        <DComposeItem dGray>.com</DComposeItem>
      </DCompose>
      <br />
      <DCompose>
        <DComposeItem dGray>$</DComposeItem>
        <DInputAffix dNumber>
          <DInput placeholder="Number" max={1000} min={10} step={10} />
        </DInputAffix>
      </DCompose>
    </>
  );
}
```
