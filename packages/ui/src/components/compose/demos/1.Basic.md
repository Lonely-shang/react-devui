# zh-Hant

最简单的用法。

```tsx
import { useState } from 'react';

import { DCompose, DButton, DIcon, DInput, DSelect, DCascader } from '@react-devui/ui';

export default function Demo() {
  const searchIcon = <DIcon dName="search"></DIcon>;
  const [options] = useState(() =>
    Array(100)
      .fill(0)
      .map((item, index) => ({ dLabel: `react-devui ${index + 1}`, dValue: index + 1, dDisabled: index === 3 }))
  );
  const [cascaderOptions] = useState(() =>
    Array(3)
      .fill(0)
      .map((item, index1) => ({
        dLabel: `Option ${index1 + 1}`,
        dValue: index1 + 1,
        dChildren: Array(5)
          .fill(0)
          .map((item, index2) => ({
            dLabel: `Option ${index1 + 1}${index2 + 1}`,
            dValue: index2 + 1,
            dDisabled: index2 === 3,
            dChildren: Array(20)
              .fill(0)
              .map((item, index3) => ({
                dLabel: `Option ${index1 + 1}${index2 + 1}${index3 + 1}`,
                dValue: index3 + 1,
              })),
          })),
      }))
  );

  return (
    <>
      <DCompose>
        <DInput placeholder="Search" />
        <DButton dIcon={searchIcon}></DButton>
      </DCompose>
      <br />
      <DCompose>
        <DSelect dOptions={options} dPlaceholder="Search"></DSelect>
        <DButton dIcon={searchIcon}></DButton>
      </DCompose>
      <br />
      <DCompose>
        <DCascader dOptions={cascaderOptions} dPlaceholder="Search"></DCascader>
        <DButton dIcon={searchIcon}></DButton>
      </DCompose>
    </>
  );
}
```
